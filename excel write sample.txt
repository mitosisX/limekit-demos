xlsxwriter in lua


-- Create a new workbook and add a worksheet
workbook = Workbook(misc('demo.xlsx'))
worksheet = workbook:add_worksheet()

-- Add a bold format to use for cell formatting
bold_format = workbook:add_format({bold = true})

-- Add a number format for cells with money
money_format = workbook:add_format({num_format = '$#,##0'})

-- Write some data headers
worksheet:write('A1', 'Item', bold_format)
worksheet:write('B1', 'Cost', bold_format)

-- Some data we want to write to the worksheet
expenses = {
    {'Rent', 1000},
    {'Gas', 100},
    {'Food', 300},
    {'Gym', 50},
}

-- Start from the first cell below the headers
row = 1
col = 0

-- Iterate over the data and write it out row by row
for _, expense in ipairs(expenses) do
    local item, cost = table.unpack(expense)
    worksheet:write(row, col, item)
    worksheet:write(row, col + 1, cost, money_format)
    row = row + 1
end

-- Write a total using a formula
worksheet:write(row, 0, 'Total', bold_format)
worksheet:write_formula(row, 1, '=SUM(B2:B5)', money_format)

-- Close the workbook to save it
workbook:close()
